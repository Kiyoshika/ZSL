6// A class to help with printing data sets to the console
7// in a table format. All column sizes will be fixed to 15
8// characters (two tab lengths). Any strings past 16 characters
9// will have the last three characters replaced with "..."
10
11// As an aside, this is pretty messy and could definitely use some cleaning,
12// but for now I just want something that works.
13class PrintTable
14{
15 public:
16 static void print_table(DataSet<std::string> const& dataset, size_t rows, size_t column_width)
17 {
18
19 if (column_width < 5) { throw std::runtime_error("When setting table column width, please use a width of at least 5 characters."); }
20
21 // if passed rows exceeds the total number of rows we have, then limit it
22 // by the total rows we have. Otherwise, continue with the passed value
23 rows = rows > dataset.count_rows() ? dataset.count_rows() : rows;
24
25 // print top horizontal line
26
27 for (auto const& col : dataset.get_column_names())
28 {
29 print_horizontal_line(column_width);
30 }
31 std::cout << "+\n";
32
33 // print headers
34
35 print_record(dataset.get_column_names(), column_width, true);
36
37 // print body (will not need the top print_horizontal_line)
38 for (size_t row = 0; row < rows; ++row)
39 {
40 print_record(dataset.get_row(row), column_width);
41 }
42
43 // print final bottom horizontal line
44 for (auto const& col : dataset.get_column_names())
45 {
46 print_horizontal_line(column_width);
47 }
48 std::cout << "+\n";
49 }

54 static void print_horizontal_line(size_t column_width)
55 {
56 std::string prefix = "+";
57 std::string line(column_width + 2, '-');
58 std::cout << prefix << line;
59 }
60
61 // takes a vector of values to write as a record (row) using
62 // vertical dividers (|)
63 static void print_record(std::vector<std::string> const& content, size_t column_width, bool print_horizontal = false)
64 {
65 std::string text;
66 std::cout << "| ";
67 for (auto const& value : content)
68 {
69 text = value;
70 if (text.length() > column_width)
71 {

73 text += "...";
74 std::cout << text << " | ";
75 }
76 else if (text.length() < column_width)
77 {
78 uint8_t whitespace = column_width - text.length();
79 std::string whitespace_str(whitespace, ' ');
80 text += whitespace_str;
81 std::cout << text << " | ";
82 }
83 else
84 {
85 std::cout << text << "| ";
86 }
87 }
88
89 std::cout << "\n";
90
91 if (print_horizontal)
92 {
93 for (auto const& value : content)
94 {
95 print_horizontal_line(column_width);
96 }
